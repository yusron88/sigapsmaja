// ============================================================ // SIGAP - Sistem Penilaian dan Penegakan Disiplin // SMA Negeri Jatirogo // ============================================================ const SPREADSHEET_ID = '1ejr5gZWFDKzookF8m8QTg2QekD-7NkxyZ8h3m9AO2wM'; const CONFIG = { sheets: { users: 'Users', students: 'Students', classes: 'Classes', categories: 'Categories', violations: 'Violations', achievements: 'Achievements' } }; // ============================================================ // DO GET - Entry point web app // ============================================================ function doGet(e) { return HtmlService.createTemplateFromFile('Index') .evaluate() .setTitle('SIGAP - SMA Negeri Jatirogo') .addMetaTag('viewport', 'width=device-width, initial-scale=1') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); } // ============================================================ // UTILITY - Get sheet by name // ============================================================ function getSheet(name) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); } return sheet; } // ============================================================ // AUTHENTICATION // ============================================================ function doLogin(username, password) { const sheet = getSheet(CONFIG.sheets.users); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { const row = data[i]; if (row[0] === username && row[1] === password) { return { success: true, role: row[2] || 'siswa', username: row[0], nama: row[3] || row[0] }; } } return { success: false, message: 'Username atau password salah' }; } // ============================================================ // STUDENT DATA // ============================================================ function getStudents() { const sheet = getSheet(CONFIG.sheets.students); const data = sheet.getDataRange().getValues(); const headers = data[0] || []; const result = []; for (let i = 1; i < data.length; i++) { const obj = {}; headers.forEach((h, idx) => { obj[h] = data[i][idx] || ''; }); result.push(obj); } return result; } function getStudentByNis(nis) { const students = getStudents(); return students.find(s => s.NIS == nis) || null; } function getStudentsByClass(className) { const students = getStudents(); return students.filter(s => s.Kelas === className); } function saveStudent(data) { const sheet = getSheet(CONFIG.sheets.students); const headers = ['NIS', 'Nama', 'Kelas', 'JK']; const existing = getStudents(); const idx = existing.findIndex(s => s.NIS == data.NIS); if (idx >= 0) { const row = idx + 2; sheet.getRange(row, 1, 1, headers.length).setValues([ [data.NIS, data.Nama, data.Kelas, data.JK] ]); } else { sheet.appendRow([data.NIS, data.Nama, data.Kelas, data.JK]); } return { success: true }; } function deleteStudent(nis) { const sheet = getSheet(CONFIG.sheets.students); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == nis) { sheet.deleteRow(i + 1); return { success: true }; } } return { success: false, message: 'Siswa tidak ditemukan' }; } // ============================================================ // CLASS DATA // ============================================================ function getClasses() { const sheet = getSheet(CONFIG.sheets.classes); const data = sheet.getDataRange().getValues(); const result = []; for (let i = 1; i < data.length; i++) { if (data[i][0]) result.push(data[i][0]); } return result; } function saveClass(className) { const sheet = getSheet(CONFIG.sheets.classes); const existing = getClasses(); if (!existing.includes(className)) { sheet.appendRow([className]); } return { success: true }; } function deleteClass(className) { const sheet = getSheet(CONFIG.sheets.classes); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === className) { sheet.deleteRow(i + 1); return { success: true }; } } return { success: false }; } // ============================================================ // CATEGORIES (Violation & Achievement) // ============================================================ function getCategories(type) { const sheet = getSheet(CONFIG.sheets.categories); const data = sheet.getDataRange().getValues(); const result = []; for (let i = 1; i < data.length; i++) { if (data[i][0] === type && data[i][1]) { result.push({ nama: data[i][1], poin: data[i][2] || 0 }); } } return result; } function saveCategory(type, nama, poin) { const sheet = getSheet(CONFIG.sheets.categories); sheet.appendRow([type, nama, poin]); return { success: true }; } function deleteCategory(type, nama) { const sheet = getSheet(CONFIG.sheets.categories); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === type && data[i][1] === nama) { sheet.deleteRow(i + 1); return { success: true }; } } return { success: false }; } // ============================================================ // VIOLATIONS & ACHIEVEMENTS (Poin entries) // ============================================================ function savePoin(data) { const sheetName = data.type === 'pelanggaran' ? CONFIG.sheets.violations : CONFIG.sheets.achievements; const sheet = getSheet(sheetName); // Auto-generate ID const lastRow = sheet.getLastRow(); const id = lastRow === 0 ? 1 : sheet.getRange(lastRow, 1).getValue() + 1; const row = [ id, data.tanggal || new Date().toISOString().split('T')[0], data.nis, data.namaSiswa, data.kelas, data.kategori, data.poin || 0, data.keterangan || '', data.petugas || '', data.fotoUrl || '', new Date().toISOString() ]; sheet.appendRow(row); return { success: true, id: id }; } function getPoinByNis(nis) { const violations = getViolations(); const achievements = getAchievements(); const semua = [...violations, ...achievements]; return semua.filter(item => item.NIS == nis); } function getViolations() { return getPoinData(CONFIG.sheets.violations, 'pelanggaran'); } function getAchievements() { return getPoinData(CONFIG.sheets.achievements, 'prestasi'); } function getPoinData(sheetName, type) { const sheet = getSheet(sheetName); const data = sheet.getDataRange().getValues(); if (data.length < 2) return []; const headers = data[0]; const result = []; for (let i = 1; i < data.length; i++) { const obj = {}; headers.forEach((h, idx) => { obj[h] = data[i][idx] || ''; }); obj.type = type; result.push(obj); } return result; } function getAllPoin() { return { violations: getViolations(), achievements: getAchievements() }; } // ============================================================ // DASHBOARD STATISTICS // ============================================================ function getDashboardStats() { const students = getStudents(); const violations = getViolations(); const achievements = getAchievements(); // Total poin per siswa (kompensasi: prestasi mengurangi pelanggaran) const poinMap = {}; students.forEach(s => { poinMap[s.NIS] = { nama: s.Nama, kelas: s.Kelas, totalPelanggaran: 0, totalPrestasi: 0, total: 0 }; }); violations.forEach(v => { if (poinMap[v.NIS]) { poinMap[v.NIS].totalPelanggaran += Number(v.Poin) || 0; } }); achievements.forEach(a => { if (poinMap[a.NIS]) { poinMap[a.NIS].totalPrestasi += Number(a.Poin) || 0; } }); const result = []; Object.keys(poinMap).forEach(nis => { const d = poinMap[nis]; d.total = d.totalPelanggaran - d.totalPrestasi; if (d.total > 0 || d.totalPelanggaran > 0 || d.totalPrestasi > 0) { result.push(d); } }); // Statistik per kelas const kelasStats = {}; result.forEach(d => { if (!kelasStats[d.kelas]) { kelasStats[d.kelas] = { total: 0, count: 0, max: 0 }; } kelasStats[d.kelas].total += d.total; kelasStats[d.kelas].count += 1; if (d.total > kelasStats[d.kelas].max) kelasStats[d.kelas].max = d.total; }); return { siswa: result, perKelas: kelasStats, totalViolations: violations.length, totalAchievements: achievements.length, siswaTerbanyak: result.sort((a, b) => b.total - a.total).slice(0, 5) }; } // ============================================================ // USERS MANAGEMENT (Admin only) // ============================================================ function getUsers() { const sheet = getSheet(CONFIG.sheets.users); const data = sheet.getDataRange().getValues(); const result = []; for (let i = 1; i < data.length; i++) { if (data[i][0]) { result.push({ username: data[i][0], password: data[i][1] || '', role: data[i][2] || 'siswa', nama: data[i][3] || data[i][0] }); } } return result; } function saveUser(data) { const sheet = getSheet(CONFIG.sheets.users); const users = getUsers(); const idx = users.findIndex(u => u.username === data.username); if (idx >= 0) { const row = idx + 2; sheet.getRange(row, 1, 1, 4).setValues([ [data.username, data.password, data.role, data.nama] ]); } else { sheet.appendRow([data.username, data.password, data.role, data.nama]); } return { success: true }; } function deleteUser(username) { const sheet = getSheet(CONFIG.sheets.users); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === username) { sheet.deleteRow(i + 1); return { success: true }; } } return { success: false }; } // ============================================================ // FILE UPLOAD - Upload foto ke Google Drive // ============================================================ function uploadFile(base64Data, fileName) { try { const folderName = 'SIGAP_Photos'; let folder = DriveApp.getFoldersByName(folderName); if (!folder.hasNext()) { folder = DriveApp.createFolder(folderName); } else { folder = folder.next(); } const blob = Utilities.newBlob( Utilities.base64Decode(base64Data.split(',')[1]), 'image/jpeg', fileName ); const file = folder.createFile(blob); const fileId = file.getId(); const url = 'https://drive.google.com/uc?export=view&id=' + fileId; return { success: true, url: url, fileId: fileId }; } catch (e) { return { success: false, error: e.toString() }; } } // ============================================================ // REPORT - Generate PDF // ============================================================ function generateReport(filterKelas) { const stats = getDashboardStats(); let data = stats.siswa; if (filterKelas && filterKelas !== 'Semua') { data = data.filter(d => d.kelas === filterKelas); } // Buat HTML untuk PDF const html = `

NoNamaKelasPoin PelanggaranPoin PrestasiTotal PoinStatus
${i+1} ${d.nama} ${d.kelas} ${d.totalPelanggaran} ${d.totalPrestasi} ${d.total} ${d.total >= 100 ? '⚠️ Segera ke BK' : d.total > 50 ? '⚠️ Lapor Wali Kelas' : 'Normal'}

Statistik

Total Siswa${data.length}
Total Pelanggaran${stats.totalViolations}

Dicetak dari SIGAP - SMA Negeri Jatirogo
`; const blob = Utilities.newBlob(html, 'text/html', 'report.html'); const pdfBlob = blob.getAs('application/pdf'); const file = DriveApp.createFile(pdfBlob).setName('Laporan_SIGAP_' + new Date().toISOString().split('T')[0] + '.pdf'); return { success: true, url: file.getUrl(), fileId: file.getId() }; } // ============================================================ // INIT - Setup default sheets & data jika kosong // ============================================================ function initSystem() { // Users let sheet = getSheet(CONFIG.sheets.users); if (sheet.getLastRow() === 0) { sheet.appendRow(['Username', 'Password', 'Role', 'Nama']); sheet.appendRow(['admin', 'admin123', 'admin', 'Administrator']); sheet.appendRow(['siswa1', 'siswa123', 'siswa', 'Siswa Contoh']); } // Classes sheet = getSheet(CONFIG.sheets.classes); if (sheet.getLastRow() === 0) { sheet.appendRow(['Kelas']); ['X-A', 'X-B', 'XI-A', 'XI-B', 'XII-A', 'XII-B'].forEach(k => sheet.appendRow([k])); } // Students sheet = getSheet(CONFIG.sheets.students); if (sheet.getLastRow() === 0) { sheet.appendRow(['NIS', 'Nama', 'Kelas', 'JK']); sheet.appendRow(['001', 'Budi Santoso', 'X-A', 'L']); sheet.appendRow(['002', 'Ani Wijaya', 'X-A', 'P']); } // Categories sheet = getSheet(CONFIG.sheets.categories); if (sheet.getLastRow() === 0) { sheet.appendRow(['Type', 'Nama', 'Poin']); const cats = [ ['pelanggaran', 'Terlambat', 10], ['pelanggaran', 'Bolos', 59], ['pelanggaran', 'Merokok', 75], ['pelanggaran', 'Berkelahi', 100], ['prestasi', 'Juara Kelas', 30], ['prestasi', 'Juara Lomba', 50], ['prestasi', 'Juara Nasional', 75] ]; cats.forEach(c => sheet.appendRow(c)); } // Violations & Achievements getSheet(CONFIG.sheets.violations); getSheet(CONFIG.sheets.achievements); return { success: true, message: 'Sistem SIGAP siap digunakan!' }; }
